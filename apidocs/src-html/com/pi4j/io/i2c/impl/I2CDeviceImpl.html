030import com.pi4j.io.i2c.I2CBus;
031import com.pi4j.io.i2c.I2CDevice;
032import com.pi4j.jni.I2C;
033
034import java.io.IOException;
035import java.util.concurrent.locks.Lock;
036import java.util.concurrent.locks.ReentrantLock;
037
038/**
039 * Implementation of i2c device. This class only holds reference to i2c bus (so it can use its handle) and
040 * device address.
041 * 
042 * @author Daniel Sendula
043 *
044 */
045public class I2CDeviceImpl implements I2CDevice {
046
047    /** Reference to i2c bus */
048    private I2CBus bus;
049    
050    /** I2c device address */
051    private int deviceAddress;
052
053    private final static Lock lock = new ReentrantLock( true );
054    
055    /**
056     * Constructor.
057     * 
058     * @param bus i2c bus
059     * @param address i2c device address
060     */
061    public I2CDeviceImpl(I2CBus bus, int address) {
062        this.bus = bus;
063        this.deviceAddress = address;
064    }
065
066    /**
067     * This method writes one byte to i2c device. 
068     * 
069     * @param data byte to be written
070     * 
071     * @throws IOException thrown in case byte cannot be written to the i2c device or i2c bus
072     */
073    @Override
074    public void write(byte data) throws IOException {
075        lock.lock();
076        try {
077            int ret = I2C.i2cWriteByteDirect(bus.getFileDescriptor(), deviceAddress, data);
078            if (ret < 0) {
079                throw new IOException("Error writing to " + makeDescription() + ". Got " + ret + ".");
080            }
081        } finally {
082            lock.unlock();
083        }
084    }
085
086    /**
087     * This method writes several bytes to the i2c device from given buffer at given offset.
088     * 
089     * @param buffer buffer of data to be written to the i2c device in one go
090     * @param offset offset in buffer 
091     * @param size number of bytes to be written 
092     * 
093     * @throws IOException thrown in case byte cannot be written to the i2c device or i2c bus
094     */
095    @Override
096    public void write(byte[] buffer, int offset, int size) throws IOException {
097        lock.lock();
098        try {
099            int ret = I2C.i2cWriteBytesDirect(bus.getFileDescriptor(), deviceAddress, size, offset, buffer);
100            if (ret < 0) {
101                throw new IOException("Error writing to " + makeDescription() + ". Got " + ret + ".");
102            }
103        } finally {
104            lock.unlock();
105        }
106    }
107
108    /**
109     * This method writes one byte to i2c device. 
110     * 
111     * @param address local address in the i2c device
112     * @param data byte to be written
113     * 
114     * @throws IOException thrown in case byte cannot be written to the i2c device or i2c bus
115     */
116    @Override
117    public void write(int address, byte data) throws IOException {
118        lock.lock();
119        try {
120            int ret = I2C.i2cWriteByte(bus.getFileDescriptor(), deviceAddress, address, data);
121            if (ret < 0) {
122                throw new IOException("Error writing to " + makeDescription(address) + ". Got " + ret + ".");
123            }
124        } finally {
125            lock.unlock();
126        }
127    }
128
129    /**
130     * This method writes several bytes to the i2c device from given buffer at given offset.
131     * 
132     * @param address local address in the i2c device
133     * @param buffer buffer of data to be written to the i2c device in one go
134     * @param offset offset in buffer 
135     * @param size number of bytes to be written 
136     * 
137     * @throws IOException thrown in case byte cannot be written to the i2c device or i2c bus
138     */
139    @Override
140    public void write(int address, byte[] buffer, int offset, int size) throws IOException {
141        lock.lock();
142        try {
143            int ret = I2C.i2cWriteBytes(bus.getFileDescriptor(), deviceAddress, address, size, offset, buffer);
144            if (ret < 0) {
145                throw new IOException("Error writing to " + makeDescription(address) + ". Got " + ret + ".");
146            }
147        } finally {
148            lock.unlock();
149        }
150    }
151
152    /**
153     * This method reads one byte from the i2c device.
154     * Result is between 0 and 255 if read operation was successful, else a negative number for an error.
155     *
156     * @return byte value read: positive number (or zero) to 255 if read was successful. Negative number if reading failed.
157     * 
158     * @throws IOException thrown in case byte cannot be read from the i2c device or i2c bus
159     */
160    @Override
161    public int read() throws IOException {
162        lock.lock();
163        try {
164            int ret = I2C.i2cReadByteDirect(bus.getFileDescriptor(), deviceAddress);
165            if (ret < 0) {
166                throw new IOException("Error reading from " + makeDescription() + ". Got " + ret + ".");
167            }
168            return ret;
169        } finally {
170            lock.unlock();
171        }
172    }
173
174    /**
175     * <p>This method reads bytes from the i2c device to given buffer at asked offset. </p>
176     * 
177     * <p>Note: Current implementation calls {@link #read(int)}. That means for each read byte 
178     * i2c bus will send (next) address to i2c device.
179     * </p>
180     * 
181     * @param buffer buffer of data to be read from the i2c device in one go
182     * @param offset offset in buffer 
183     * @param size number of bytes to be read 
184     * 
185     * @return number of bytes read
186     * 
187     * @throws IOException thrown in case byte cannot be read from the i2c device or i2c bus
188     */
189    @Override
190    public int read(byte[] buffer, int offset, int size) throws IOException {
191        lock.lock();
192        try {
193            // It doesn't work for some reason.
194            int ret = I2C.i2cReadBytesDirect(bus.getFileDescriptor(), deviceAddress, size, offset, buffer);
195            if (ret < 0) {
196                throw new IOException("Error reading from " + makeDescription() + ". Got " + ret + ".");
197            }
198            return ret;
199        } finally {
200            lock.unlock();
201        }
202    }
203
204    /**
205     * This method reads one byte from the i2c device.
206     * Result is between 0 and 255 if read operation was successful, else a negative number for an error.
207     *
208     * @param address local address in the i2c device
209     * @return byte value read: positive number (or zero) to 255 if read was successful. Negative number if reading failed.
210     * 
211     * @throws IOException thrown in case byte cannot be read from the i2c device or i2c bus
212     */
213    @Override
214    public int read(int address) throws IOException {
215        lock.lock();
216        try {
217            int ret = I2C.i2cReadByte(bus.getFileDescriptor(), deviceAddress, address);
218            if (ret < 0) {
219                throw new IOException("Error reading from " + makeDescription(address) + ". Got " + ret + ".");
220            }
221            return ret;
222        } finally {
223            lock.unlock();
224        }
225    }
226
227    /**
228     * <p>This method reads bytes from the i2c device to given buffer at asked offset. </p>
229     * 
230     * <p>Note: Current implementation calls {@link #read(int)}. That means for each read byte 
231     * i2c bus will send (next) address to i2c device.
232     * </p>
233     * 
234     * @param address local address in the i2c device
235     * @param buffer buffer of data to be read from the i2c device in one go
236     * @param offset offset in buffer 
237     * @param size number of bytes to be read 
238     * 
239     * @return number of bytes read
240     * 
241     * @throws IOException thrown in case byte cannot be read from the i2c device or i2c bus
242     */
243    @Override
244    public int read(int address, byte[] buffer, int offset, int size) throws IOException {
245        lock.lock();
246        try {
247            // It doesn't work for some reason.
248            int ret = I2C.i2cReadBytes(bus.getFileDescriptor(), deviceAddress, address, size, offset, buffer);
249            if (ret < 0) {
250                throw new IOException("Error reading from " + makeDescription(address) + ". Got " + ret + ".");
251            }
252            return ret;
253        } finally {
254            lock.unlock();
255        }
256    }
257
258    /**
259     * This method writes and reads bytes to/from the i2c device in a single method call
260     *
261     * @param writeBuffer buffer of data to be written to the i2c device in one go
262     * @param writeOffset offset in write buffer
263     * @param writeSize number of bytes to be written from buffer
264     * @param readBuffer buffer of data to be read from the i2c device in one go
265     * @param readOffset offset in read buffer
266     * @param readSize number of bytes to be read
267     *
268     * @return number of bytes read
269     *
270     * @throws IOException thrown in case byte cannot be read from the i2c device or i2c bus
271     */
272    @Override
273    public int read(byte[] writeBuffer, int writeOffset, int writeSize, byte[] readBuffer, int readOffset, int readSize) throws IOException {
274        lock.lock();
275        try {
276            int ret = I2C.i2cWriteAndReadBytes(bus.getFileDescriptor(), deviceAddress, writeSize, writeOffset, writeBuffer, readSize, readOffset, readBuffer);
277            if (ret < 0) {
278                throw new IOException("Error reading from " + makeDescription() + ". Got " + ret + ".");
279            }
280            return ret;
281        } finally {
282            lock.unlock();
283        }
284    }
285
286    /**
287     * This helper method creates a string describing bus file name and device address (in hex).
288     * 
289     * @return string with all details
290     */
291    protected String makeDescription() {
292        return bus.getFileName() + " at address 0x" + Integer.toHexString(deviceAddress);
293    }
294    
295    /**
296     * This helper method creates a string describing bus file name, device address (in hex)
297     * and local i2c address.
298     * 
299     * @param address local address in i2c device
300     * @return string with all details
301     */
302    protected String makeDescription(int address) {
303        return bus.getFileName() + " at address 0x" + Integer.toHexString(deviceAddress) 
304                + " to address 0x" + Integer.toHexString(address);
305    }
306    
307}
